	e.g-> cd "C:\Users\mahbisht\Documents\MyPractice\Git Notes"

	e.g-> git init

	git add remote <remote-name> <path>
	e.g-> git add remote origin "https://github.com/MahimaBisht/GitHub-notes.git"

	e.g-> git branch -m master
